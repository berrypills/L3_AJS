
function getSortedPosts(callback) {
  fetch('https://jsonplaceholder.typicode.com/posts')
    .then(response => response.json())
    .then(data => {
      const sorted = data.sort((a, b) => b.title.length - a.title.length);
      callback(sorted);
    });
}

function getSortedComments(callback) {
  fetch('https://jsonplaceholder.typicode.com/comments')
    .then(response => response.json())
    .then(data => {
      const sorted = data.sort((a, b) => a.name.localeCompare(b.name));
      callback(sorted);
    });
}

getSortedPosts(posts => console.log('Sorted Posts:', posts));
getSortedComments(comments => console.log('Sorted Comments:', comments));
